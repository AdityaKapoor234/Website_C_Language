Q4) C PROGRAM TO REMOVE ALL OCCURRENCE OF WORD “The” AND ”the” FROM AN INPUT STRING. FOR EXAMPLE:

INPUT: The Dhillon Theatre is now the Fun Republic.
OUTPUT: Dhillon atre is now Fun Republic.

ANS) 
#include < stdio.h >
#include < conio.h >
void main()
{
int i,j,k,n,x,x2;
char str[100];
clrscr();
printf("THIS IS A PROGRAM TO REMOVE THE OCCURRENCES OF \"The\" & \"the\" FROM THE INPUT STRING\n\nEnter The  Value Of String \n");
/*scanning the value entered in string*/
gets(str);
n=strlen(str);

/*removing the occurrences of "The" & "the"*/
for (i=0 , j=1 , k=2 ;  k < n ; i++ , j++ , k++)
{
	if ((str[i]=='T' && str[j]=='h' && str[k]=='e') || (str[i]=='t' && str[j]=='h' && str[k]=='e'))
	{
		for (x=i , x2=(k+1) ;  x2 < n ; x++ , x2++)
		{
			str[x]=str[x2];
		}
		str[n-1]=NULL;
		str[n-2]=NULL;
		str[n-3]=NULL;
	}
}

/*display the result*/
printf("\n\nRESULT)\n");
puts(str);
getch();
}

Q8) FUNCTION char* stuff(char* s1,char* s2,int sp, int rp) TO STUFF STRING s2 IN STRING s1 AT POSITION sp, REPLACING RP NUMBER OF CHARACTERS (rp MAY BE ZERO).
ANS) 
#include < stdio.h >
#include < conio.h >
void main()
{
int sp,rp;
char *s1, *s2;
void stuff(char *s1,char *s2,int sp,int rp);
clrscr();
printf("THIS IS A PROGRAM TO STUFF SOME CHARACTER OF SECOND STRING IN FIRST STRING STARTING FROM A SPECIFIED POSITION\n\nEnter First String ");
gets(s1);
printf("\nEnter Second String ");
gets(s2);
printf("\nEnter The Position Where You Want To Enter The Some Specific Character Of First String ");
scanf("%d",&sp);
printf("\nEnter How Many Character Of Second String You Want To Enter In The First String ");
scanf("%d",&rp);
stuff(s1,s2,sp,rp);
getch();
}

void stuff(char *s1,char *s2,int sp,int rp)
{
int i,j,ch;
char str[100];
ch=sp+rp;

/*entering the values of first string(s1) into a new string(str)*/
for (i=sp , j=0 ; i < strlen(s1) ; i++ , j++)
	str[j]=s1[i];
str[j]='\0';


/*shifting the values of first string(s1) from a specific place to some place ahead*/
for (i=ch , j=0 ; j < strlen(str) ; i++ , j++)
	s1[i]=str[j];

/*entering some values of second string into first string*/
for (i=sp , j=0 ; i < ch , j < rp ; i++ , j++)
	s1[i]=s2[j];
printf("\n\nAns) %s",s1);
}
