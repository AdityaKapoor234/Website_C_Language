Q1) C PROGRAM TO CREATE A CIRCULAR QUEUE FOR INTEGER DATA TYPE AND THEN DISPLAY THE VALUES OF CIRCULAR QUEUE
ANS) 
#include < stdio.h >
#include < conio.h >
struct cir_queue
{
int a[10];
int front;
int rear;
}q;

void main()
{
void create();
void display();
clrscr();
printf("THIS IS A PROGRAM TO CREATE AND DISPLAY CIRCULAR QUEUE\n\n");
create();
display();
getch();
}

void create()
{
int n;
q.front=-1;
q.rear=-1;
printf("Enter The Values & Enter 0 To Exit\n");
scanf("%d",&n);
while (n!=0)
{
	if (q.front==-1)
	{
		q.front++;
		q.rear++;
		q.a[q.front]=n;
	}
	else if (q.rear==9 && q.front!=0)
	{
		q.rear=0;
		q.a[q.rear]=n;
	}
	else if (q.rear!=9)
	{
		q.rear++;
		q.a[q.rear]=n;
	}
	else
	{
		printf("Cannot Add More Elements\n");
		break;
	}
	if ((q.front!=q.rear+1) && !(q.front==0 && q.rear==9))
		scanf("%d",&n);
}
printf("End Of Queue\n\n");
}

void display()
{
int i;
printf("DISPLAY\n");
for (i=q.front ; i<=q.rear ; i++)
{
	printf("%d\t",q.a[i]);
	if ((q.front!=0) && i==9)
		i=0;
}
}
