Q1) C PROGRAM TO CREATE A LINKED LIST FOR INTEGER DATA TYPE AND THEN DISPLAY THE VALUES OF LINKED ELEMENTS
ANS) 
#include < stdio.h >
#include < conio.h >
struct node
{
int data;
struct node *next;
}*start=NULL,*last=NULL;

void main()
{

void create();
void display();

{
	create();
	display();
}

}

{
int n=1;
//function to create a linked list
printf("Enter Numbers & Enter 0 To Exit\n");
while (n!=0)
{
	scanf("%d",&n);
	if (n==0)
	{
		printf("End Of List\n\n");
		break;
	}
	else
	{
		struct node *node1;
		node1=(struct node*)malloc(sizeof(struct node));
		node1->data=n;
		node1->next=NULL;

		{
			start=node1;
			last=node1;
		}
		else
		{
			last->next=node1;
			last=last->next;
		}
	}

}
