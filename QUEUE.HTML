Q1) C PROGRAM TO CREATE A QUEUE FOR INTEGER DATA TYPE AND THEN DISPLAY THE VALUES OF QUEUE
ANS) 
#include < stdio.h >
#include < conio.h >
struct queue
{
int a[10];
int front;
int real;
}q;

void main()
{
void create();
void display();
clrscr();
printf("THIS IS A PROGRAM TO CREATE & DISPLAY QUEUE\n\n");
create();
display();
getch();
}

void create()
{
int n;
q.front=-1;
q.real=-1;
printf("Enter The Values Of Queue & Enter 0 To Exit\n");
scanf("%d",&n);
while (n!=0)
{
	if  (q.front==-1)
	{
		q.front++;
		q.real++;
		q.a[q.front]=n;
	}
	else
	{
		if (q.real==9)
		{
			printf("Queue Is Full\n\n");
			break;
		}
		else
		{
			q.real++;
			q.a[q.real]=n;
		}
	}
	if (q.real<9)
		scanf("%d",&n);
	if (n==0)
		printf("END\n\n");
}
}

void display()
{
int i;
printf("QUEUE\n");
if (q.front==-1)
	printf("EMPTY");
else
{
	for (i=q.front ; i<=q.real ; i++)
		printf("%d\t",q.a[i]);
}
}
