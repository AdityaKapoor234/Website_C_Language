Q2) C PROGRAM TO CONCATENATE TWO STRING WITHOUT USING STRING FUNCTIONS
ANS) 
#include < stdio.h >
#include < conio.h >
void main()
{
char a[100],b[100];
int i,a_len=0,b_len=0,c,j;
clrscr();
printf("THIS IS A PROGRAM TO CONCATENATE TWO DIFFERENT VARIABLES\n\nEnter Your First Name ");
gets(a);
printf("Enter Your Last Name ");
gets(b);
/*calculating the length of first string(a)*/
for (i=0;a[i]!='\0';i++)
	a_len++;

/*calculating the length of second string(a)*/
for (i=0;b[i]!='\0';i++)
	b_len++;

/*calculating the total length of first string(a) after concatenation*/
c=a_len+b_len;

/*concatenating the value of second string into first string*/
for (i=a_len , j=0 ; i < c , j < b_len ; i++ , j++)
	a[i]=b[j];

printf("\nYour Concatenated Value is %s %s",a);
getch();
}

Q7) C PROGRAM TO INPUT A STRING FROM THE USER, FIND THE STRING A PALINDROME OR NOT WITHOUT USING STRING FUNCTIONS
ANS) 
1ST WAY
#include < stdio.h >
#include < conio.h >
void main()
{
char a[100];
int i,j,n,len=0;
clrscr();
printf("PALINDROM\n\nEnter a Word ");
gets(a);
n=0;
while (a[n]!='\0')
{
	len++;
	n++;
}
for (i=len-1 , j=0 ; i>=0 , j < len ; i-- , j++)
{
	if (a[j]!=a[i])
	{
		printf("Not A Palindrome");
		getch();
		exit(0);
	}
}
printf("It's a Palindrom");
getch();
}
