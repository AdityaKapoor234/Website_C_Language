Q5) C PROGRAM TO EVALUATE CO-SINE SERIES

cos x= 1- x2/2! + x4/4! - x6/6! + x8/8! - ...N
ANS) 
#include < stdio.h >
#include < conio.h >
#include < math.h >
void main()
{
int x,p,i=0,p2=0,sine=1;
int facto(int p2);
double m=0,a;
clrscr();
printf("CO-SINE SERIES\n\nEnter Value ");
scanf("%d",&x);
printf("Enter Number of Terms ");
scanf("%d",&p);
while (i < p)
{
	a=pow(x,p2);
	m+=sine*(a/facto(p2));
	sine*=-1;
	p2+=2;
	i++;
}
printf("Ans) %.2lf",m);
getch();
}

int facto(int p2)
{
int fact=1;
while (p2>0)
{
	fact*=p2;
	p2--;
}
return fact;
}

{

Q17) C PROGRAM TO COUNT THE NUMBER OF LINES, WORDS & CHARACTER ENTERED BY USER IN THE PROGRAM
ANS) 
#include < stdio.h >
#include < conio.h >
void main()
{
char a[500],ch;
int i,li=0,wo=0,cha=0;
clrscr();
printf("THIS IS A PROGRAM TO DISPLAY THE NUMBER OF LINES, WORDS AND CHARACTERS ENTERED  BY THE USER\n\nEnter String\n\n");
/*lines being scaned in string*/
for (i=0 ; (ch=getchar()) != '$' ; i++)
{
	/*scanning value*/
	a[i]=ch;

	/*counting lines*/
	if(ch=='\n')
		li++;

	/*counting characters*/
	if (ch!=' ' && ch!='\n' && ch!='\t')
		cha++;

	/*counting words*/
	if (ch==' ' || ch=='\n' || ch=='\t')
		wo++;
}
printf("\nNo. Of Characters:%d\nNo. Of Words:%d\nNo. Of Lines:%d",cha,wo,li);
getch();
}
