Q7) FUNCTION TO DETERMINE THE SUM OF THE FOLLOWING SERIES:

S=1-3+5-7+….N READ THE VALUE OF N FROM THE USER
ANS)

#include < conio.h >

{
int n;
int ser(int n);
clrscr();
printf("THIS PROGRAM IS TO EVALUATE THE SERIES [S=1-3+5-7+.....+N]\n\nEnter the Value of N ");
scanf("%d",&n);
printf("Ans) %d",ser(n));
getch();
}

int ser(int n)
{
int ser=1,s=0,i,e=1;
for (i=0 ; i < n ; i++)
{
	s+=ser*e;
	ser*=-1;
	e+=2;
}
return s;
}

Q11) C FUNCTION TO COMPUTE THE AVERAGE OF EVERY THIRD INTEGER LYING BETWEEN 1 AND 200
ANS) 
#include < stdio.h >
#include < conio.h >
void main()
{
int i,j,sum,avg,first=1,last=3;
clrscr();
printf("THIS IS A PROGRAM TO CALCULATE THE AVERAGE OF EVERY THIRD INTEGER LYING BETWEEN 1 TO 200\n\nRange\t\tAverage\n");
for (i=1 ; i<=200 ; i+=3)
{
	sum=0;
	if (i==199)
	{
		for (j=i ; j<=(i+1) ; j++)
		{
			sum+=j;
		}
	}
	else
	{
		for (j=i ; j<=(i+2) ; j++)
		{
			sum+=j;
		}
	}
	avg=sum/3;
	printf("%d-%d:\t\t%d\n",first,last,avg);
	first+=3;
	if (last==198)
	{
		last+=2;
	}
	else
	{
		last+=3;
	}
}
getch();
}

Q13) RECURSIVE FUNCTION TO DISPLAY FIBONACCI SERIES
ANS) 
#include < stdio.h >
#include < conio.h >
void main()
{
int n,first=0,second=1;
void fibo(int n);
clrscr();
printf("THIS IS A PROGRAM O FIND THE FIRST 'N' FIBONACCI NUMBERS\n\nEnter How Many Fibonacci Number You Want To See ");
scanf("%d",&n);
if (n==0)
{
	getch();
	exit(0);
}
else if (n==1)
{
	printf("%d\n",first);
}
else if (n==2)
{
	printf("%d\n",first);
	printf("%d\n",second);
}
else if (n>2)
{
	printf("%d\n",first);
	printf("%d\n",second);
	fibo(n);
}
getch();
}


void fibo(int n)
{
static int first=0,second=1,third;
if (n>2)
{
third=first+second;
first=second;
second=third;
printf("%d\n",third);
fibo(n-1);
}
}
