{
	scanf("%d",&a[i]);
}
printf("Enter the no. to be search ");
scanf("%d",&search);
low=0;
high=n-1;
while (low<=high)
{
	mid=(low+high)/2;
	if (search==a[mid])
	{
		printf("%d element found at position %d",search,mid+1);
		getch();
		exit(0);
	}
	else if (search < a[mid])
	{
		high=mid-1;
	}
	else
	{
		low=mid+1;
	}
}
printf("element not found");
getch();
}

int equal(int **p1,int **p2)
{

{
	for (j=0;j<2;j++)
	{
		if (*(*(p1+i)+j)!=*(*(p2+i)+j))

			return 0;
		}
	}

Q26) C PROGRAM TO INPUT 3x3 MATRICES FROM THE USER AND DISPLAY THE SUM OF ROWS AND COLUMNS OF THE MATRIX

#include < stdio.h >
#include < conio.h >
void main()
{
int a[3][3],sum=0,row,col;
clrscr();
printf("THIS IS A PROGRAM TO FIND OUT THE SUM OF ROWS & COLUMNS OF MATRIX\n\nEnter The Value Of Matrix\n");
/*scaning the matrix*/
for (row=0;row<3;row++)
	for (col=0;col<3;col++)
		scanf("%d",&a[row][col]);
/*print the matrix*/
printf("MATRIX\n");
for (row=0;row<3;row++)
	printf("%d\t%d\t%d\n",a[row][0],a[row][1],a[row][2]);
printf("\n");
/*sum of row*/
for (row=0;row<3;row++)
{
	for (col=0;col<3;col++)
		sum+=a[row][col];
	printf("Sum Of Row %d: %d\n",row+1,sum);
	sum=0;
}
/*sum of column*/
for (row=0;row<3;row++)
{
	for (col=0;col<3;col++)
		sum+=a[col][row];
	printf("Sum Of Columns %d: %d\n",row+1,sum);
	sum=0;
}
getch();
}
