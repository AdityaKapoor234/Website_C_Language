#include < conio.h >
struct phone
{
char na[100];
char ad[200];
char ph[10];
};

void main()
{
int i,j,min,n;

clrscr();

if (n<=20)
{
	for (i=0 ; i < n ; i++)
	{
		printf("\nEnter Your Name ");
		scanf("%s",dir[i].na);
		printf("Enter Your Address ");
		scanf("%s",dir[i].ad);
		printf("Enter Your Phone No. ");
		scanf("%s",dir[i].ph);
	}
	for (i=0 ; i < n ; i++)
	{
		min=i;
		for (j=i+1 ; j < n ; j++)
		{
			if (stricmp(dir[min].na,dir[j].na)>0)
				min=j;
		}
		if (i!=min)
		{
			strcpy(temp,dir[min].na);
			strcpy(dir[min].na,dir[i].na);
			strcpy(dir[i].na,temp);
		}
	}
	printf("\n\nSORTED TELEPHONE DIRECTORY\n");
	printf("NAME\t\tPHONE NO.\t\tADDRESS\n");
	for (i=0 ; i < n ; i++)
	{
	printf("%s\t\t%s\t\t%s\n",dir[i].na,dir[i].ph,dir[i].ad);
	}
}
else
{
	printf("Info Of Upto 20 People Can Be Entered");
	getch();
	exit(0);
}

}

Q6) C PROGRAM WHICH ASKS THE USER TO ENTER THE NAME AND AGE OF PERSONS IN HIS GROUP. THE NUMBERS OF PERSONS IS NOT KNOWN TO THE USER IN THE BEGINNING OF THE PROGRAM. THE USER KEEPS ON ENTERING THE DATA TILL THE USER ENTERS THE AGE AS ZERO. THE PROGRAM FINALLY PRINTS THE AVERAGE AGE
ANS) 
#include < stdio.h >
#include < conio.h >
struct age
{
int age;
char na[100];
}a[100];

void main()
{
int i,e=0,sum=0,avg;
clrscr();
printf("THIS IS A PROGRAM TO INPUT THE NAME AND AGE OF PERSONS AND SHOW THE AVERAGE AGE\nNOTE:The Program Won't Be Terminated Till The User Enter The Age As 0\n");
for (i=0 ; i<=e ; i++)
{
	printf("\nEnter Your Name ");
	scanf("%s",a[i].na);
	printf("Enter Your Age ");
	scanf("%d",&a[i].age);
	sum+=a[i].age;
	if (a[i].age!=0)
	{
		e++;
	}
}
avg=sum/(e);
printf("\n\nAverage Age %d",avg);
getch();
}
