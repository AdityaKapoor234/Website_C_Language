Q10) C PROGRAM TO OBTAIN SUM OF THE FIRST 10 TERMS OF THE FOLLOWING SERIES FOR ANY POSITIVE INTEGER VALUE OF
X: X+X3/3! + X5/5! + X7/7 + ………
ANS) 
#include < stdio.h >
#include < conio.h >
void main()
{
int i,e=1,x;
double m=0,a;
int facto(int e);
clrscr();
printf("THIS IS A PROGRAM TO EVALUATE FIRST 10 TERMS OF THE SERIES [X:X+ X3/3! + X5/5! + X7/7! +...]\n\nEnter The Value Of X ");
scanf("%d",&x);
i=1;
while (i<=10)
{
	a=e*x;
	m+=a/facto(e);
	e+=2;
	i++;
}
printf("Ans) %lf",m);
getch();
}
int facto(int e)
{
int fact=1;
while (e>=1)
{
	fact*=e;
	e--;
}
return fact;
}

Q13) C PROGRAM TO CONVERT A GIVEN BINARY NUMBER TO ITS DECIMAL EQUIVALENT AND VICE VERSA
ANS)

#include < conio.h >

{

int expo(int i);

n=bi;
while (n>0)
{
	rem=n%10;
	de+=rem*expo(i);
	i++;
	n=n/10;
}

}

int expo(int i)
{
int a=0,z;
if (i>=2)
{
	for (z=2;z<=i;z++)
	{
		a+=2*2;
	}
}
else if (i==1)
{
	a=2;
}
else if (i==0)
{
	a=1;
}
return a;
}


Q14) C PROGRAM TO COUNT THE NUMBER OF LINES, WORDS & CHARACTER ENTERED BY USER IN THE PROGRAM
ANS) 
#include < stdio.h >
#include < conio.h >
void main()
{
char a[500],ch;
int i,li=0,wo=0,cha=0;
clrscr();
printf("THIS IS A PROGRAM TO DISPLAY THE NUMBER OF LINES, WORDS AND CHARACTERS ENTERED  BY THE USER\n\nEnter String\n\n");
//lines being scaned in string
i=0;
while ((ch=getchar()) != '$')
{
	/*scanning value*/
	a[i]=ch;

	/*counting lines*/
	if(ch=='\n')
		li++;

	/*counting characters*/
	if (ch!=' ' && ch!='\n' && ch!='\t')
		cha++;

	/*counting words*/
	if (ch==' ' || ch=='\n' || ch=='\t')
		wo++;
	i++;
}
printf("\nNo. Of Characters:%d\nNo. Of Words:%d\nNo. Of Lines:%d",cha,wo,li);
getch();
}
