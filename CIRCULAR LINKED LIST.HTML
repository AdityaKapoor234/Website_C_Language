CIRCULAR LINKED LIST

Q1) C PROGRAM TO CREATE A CIRCULAR LINKED LIST FOR INTEGER DATA TYPE AND THEN DISPLAY THE VALUES OF LINKED ELEMENTS
ANS) 
#include < stdio.h >
#include < conio.h >
struct node
{
int data;
struct node *next;
}*start=NULL, *last=NULL;

void main()
{
void create();
void display();
clrscr();
printf("THIS IS A PROGRAM TO CREATE & DISPLAY A CIRCULAR LINKED LIST\n\n");
create();
display();
getch();
}

void create()
{
int n=1;
printf("Enter The Values & Enter 0 To Exit\n");
while (n!=0)
{
	scanf("%d",&n);
	if (n==0)
	{
		printf("No More Value Can Be Entered\n");
		break;
	}
	else
	{
		struct node *node1;
		node1=(struct node*)malloc(sizeof(struct node));
		node1->data=n;

		if (start==NULL)
		{
			start=node1;
			last=node1;
		}
		else
		{
			last->next=node1;
			last=last->next;
		}
		node1->next=start;
	}
}
printf("End Of Linked List\n\n");
}

void display()
{
int i=1;
struct node *node1;
printf("DISPLAY\n");
node1=start;
while (i!=0)
{
	printf("%d-->",node1->data);
	node1=node1->next;
	if (node1==last)
		break;
}
printf("%d",node1->data);
}

void insert_mid()
{
int mid,total=0,n;
struct node *node1, *node2, *new_node;
printf("\n\nINSERT\nThe New Node Will Be Entered In The Middle Of The Circular Linked List\nEnter The Value Of New Node ");
scanf("%d",&n);

/*calculating total number of nodes*/
node1=start;
while (node1->next!=NULL)
{
	total++;
	node1=node1->next;
	if (node1==last)
		break;
}
total++;
mid=total/2;

/*traversal to the middle postion*/
node1=start;
while (mid>2)
{
	node1=node1->next;
	mid--;
}
node2=node1->next;

/*inserting the new node in middle*/
new_node=(struct node*)malloc(sizeof(struct node));
new_node->data=n;
node1->next=new_node;
new_node->next=node2;
printf("\n\n");
display();
}
