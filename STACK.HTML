Q1) C PROGRAM TO CREATE A STACK FOR INTEGER DATA TYPE AND THEN DISPLAY THE VALUES OF STACK
ANS) 
#include < stdio.h >
#include < conio.h >
struct stack
{
int a[10];
int top;
}s;

void main()
{
void create();
void display();
clrscr();
printf("\t\t\t\t\tSTACKS\n\n");
create();
display();
getch();
}

void create()
{
int n=1;
s.top=-1;
printf("Enter The Value Of Stack\n");
scanf("%d",&n);
while (n!=0)
{
	if (s.top<9)
	{
		s.top++;
		s.a[s.top]=n;
	}
	else
	{
		printf("OVERFLOW STACK CAN INPUT ONLY 10 VALUES\n");
		break;
	}
	if (s.top<9)
		scanf("%d",&n);
}
printf("END OF STACK\n\n");
}


void display()
{
int i;
printf("DISPLAY\n\n");
for (i=s.top ; i>=0 ; i--)
{
	printf("%d\t",s.a[i]);
}
}


Q2) C PROGRAM TO PUSH(INSERT) A ELEMENT IN STACK
ANS) 
#include < stdio.h >
#include < conio.h >
struct stack
{
int a[10];
int top;
}s;

void main()
{
void create();
void display();
void push();
clrscr();
printf("THIS IS A PROGRAM TO CREATE, DISPLAY AND PUSH A ELEMENT IN STACK\n\n");
create();
display();
push();
getch();
}

void create()
{
int n;
s.top=-1;
printf("Enter 10 Values & Enter 0 To Exit\n");
scanf("%d",&n);
while (n!=0)
{
	if (s.top<9)
	{
		s.top++;
		s.a[s.top]=n;
	}
	else
	{
		printf("Stack Overflow\n");
		break;
	}
	if (s.top!=9)
	{
		scanf("%d",&n);
	}
}
printf("End Of Stack\n\n");
}

void display()
{
int i;
i=s.top;
printf("STACK\n");
while (i>=0)
{
	printf("%d\t",s.a[i]);
	i--;
}
}

void push()
{
int n;
printf("\nPUSH");
if (s.top==9)
{
	printf("\nStack Is Already Full\n\n");
}
else
{
	s.top++;
	printf("\nEnter The New Element Of Stack ");
	scanf("%d",&n);
	s.a[s.top]=n;
	printf("\n");
	display();
}
}

{
	if (s.top<9)

		s.top++;
		s.a[s.top]=n;
