Q3) C PROGRAM TO PERFORM ARITHMETIC OPERATIONS USING SWITCH CASE
ANS) 
#include < stdio.h >
#include < conio.h >
void main()
{
char a;
int n1,n2,o;
clrscr();
printf("THIS IS A PROGRAM TO PERFORM A SPECIFIC ARITHMETIC OPERATION ON TWO NUMBERS\n\nEnter two Number with there Arithmetic Operation to be performed\nSyntax(2+3 or 2*4)\n");
scanf("%d%c%d",&n1,&a,&n2);
switch (a)
{
	case '+':
	o=n1+n2;
	printf("Your Sum is %d",o);
	break;

	case '-':
	o=n1-n2;
	printf("Your Substraction is %d",o);
	break;

	case '*':
	o=n1*n2;
	printf("Your Multiplication is %d",o);
	break;

	case '/':
	o=n1/n2;
	printf("Your Division is %d",o);
	break;

	default:
	printf("Enter a Valid Arithmetic Operation");
	break;
}
getch();
}
